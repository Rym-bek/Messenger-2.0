10   import androidx.datastore.preferences.preferencesDataStore
11   import com.example.messenger.utils.constants.Constants
12   
13   class TokenManager(private val context: Context) {
14       companion object {
15           private val TOKEN_KEY = stringPreferencesKey(Constants.TOKEN)
16       }
17   
18       private val Context.dataStore: DataStore<Preferences> by preferencesDataStore(name = Constants.DATA_STORE_NAME)
19   
20       fun getToken(): Flow<String?> {
21           return context.dataStore.data.map { preferences ->
22               preferences[TOKEN_KEY]
23           }
24       }
25   
26       suspend fun saveToken(token: String) {
27           context.dataStore.edit { preferences ->
28               preferences[TOKEN_KEY] = token
29           }
30       }
31   
32       suspend fun deleteToken() {
33           context.dataStore.edit { preferences ->
34               preferences.remove(TOKEN_KEY)
35           }
36       }
37   }
