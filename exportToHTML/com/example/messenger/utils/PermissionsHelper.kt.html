11   
12       fun hasPermissions(): Boolean
13       {
14           return if ( permissions.all { ActivityCompat.checkSelfPermission(fragment.requireContext(), it) == PackageManager.PERMISSION_GRANTED }) {
15               true
16           }
17           else {
18               requestPermission()
19               false
20           }
21       }
22   
23       private fun requestPermission() {
24           requestPermissionLauncher.launch(
25               permissions
26           )
27       }
28   
29   
30       private val requestPermissionLauncher =
31           fragment.registerForActivityResult(
32               ActivityResultContracts.RequestMultiplePermissions()
33           ) { permissions ->
34               val granted = permissions.entries.all {
35                   it.value
36               }
37               if (granted) {
38                   // Permission granted
39               }
40           }
