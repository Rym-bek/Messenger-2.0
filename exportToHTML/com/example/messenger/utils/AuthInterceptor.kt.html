3    import com.example.messenger.utils.constants.Constants
4    import kotlinx.coroutines.flow.first
5    import kotlinx.coroutines.runBlocking
6    import okhttp3.Interceptor
7    import okhttp3.Response
8    import javax.inject.Inject
9    
10   class AuthInterceptor @Inject constructor(
11       private val tokenManager: TokenManager,
12   ): Interceptor {
13       override fun intercept(chain: Interceptor.Chain): Response {
14           val token = runBlocking {
15               tokenManager.getToken().first()
16           }
17           val request = chain.request().newBuilder()
18           request.header(Constants.TOKEN_HEADER, "$token")
19           return chain.proceed(request.build())
20       }
21   }
