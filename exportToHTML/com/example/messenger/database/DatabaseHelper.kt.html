14   class DatabaseHelper(var context: Context) : SQLiteOpenHelper(context, DATABASE_NAME, null, DATABASE_VERSION) {
15   
16       companion object {
17           const val DATABASE_VERSION = 52
18           const val DATABASE_NAME = "messenger.db"
19           var DATABASE_PATH = ""
20           private var mDataBase: SQLiteDatabase? = null
21           const val TABLE_SLIDES = "slider"
22           const val TABLE_COUNTRY = "country"
23           const val COLUMN_CODE = "code"
24       }
25   
26       init {
27           DATABASE_PATH = context.applicationInfo.dataDir + "/databases/";
28       }
29   
30       fun createDataBase() {
31           if (!checkDataBase()) {
32               this.readableDatabase
33               close()
34               copyDataBase()
35           }
36       }
37   
38       private fun checkDataBase(): Boolean {
39           val dbFile = File(DATABASE_PATH + DATABASE_NAME)
40           return dbFile.exists()
41       }
42   
43       private fun copyDataBase() {
44           val mInput: InputStream = context.assets.open(DATABASE_NAME)
45           val mOutput: OutputStream = FileOutputStream(DATABASE_PATH + DATABASE_NAME)
46           val mBuffer = ByteArray(1024)
47           var mLength: Int
48           while (mInput.read(mBuffer).also { mLength = it } > 0) mOutput.write(mBuffer, 0, mLength)
49           mOutput.flush()
50           mOutput.close()
51           mInput.close()
52       }
53   
54       fun openDataBase(){
55           mDataBase = SQLiteDatabase.openDatabase(DATABASE_PATH + DATABASE_NAME, null, SQLiteDatabase.OPEN_READONLY)
56       }
57   
58       override fun close() {
59           if (mDataBase != null)
60           {
61               mDataBase!!.close()
62           }
63           super.close()
64       }
65   
66   
67       override fun onCreate(p0: SQLiteDatabase?) {
68           copyDataBase()
69       }
70   
71       override fun onUpgrade(p0: SQLiteDatabase?, p1: Int, p2: Int) {
72           onCreate(p0)
73       }
74   
75       override fun onDowngrade(db: SQLiteDatabase?, oldVersion: Int, newVersion: Int) {
76           onCreate(db)
77       }

83       fun getSliderInfo(): ArrayList<Slide> {
84           val listSlider: ArrayList<Slide> = ArrayList()
85           val query = "SELECT * FROM $TABLE_SLIDES"
86           val cursor = mDataBase!!.rawQuery(query, null)
87           cursor.moveToFirst()
88           while (!cursor.isAfterLast) {
89               val id = cursor.getInt(0)
90               val image = cursor.getString(1)
91               val title = cursor.getString(2)
92               val text = cursor.getString(3)
93               listSlider.add(Slide(id, image, title, text))
94               cursor.moveToNext()
95           }
96           cursor.close()
97           return listSlider
98       }

101      fun getCountriesInfo(): ArrayList<Country> {
102          val countryArrayList: ArrayList<Country> = ArrayList()
103          val query = "SELECT * FROM $TABLE_COUNTRY"
104          val cursor = mDataBase!!.rawQuery(query, null)
105          cursor.moveToFirst()
106          while (!cursor.isAfterLast) {
107              val id = cursor.getInt(0)
108              val flag = cursor.getString(1)
109              val name = cursor.getString(2)
110              val code = cursor.getString(3)
111              countryArrayList.add(Country(id, flag, name, code))
112              cursor.moveToNext()
113          }
114          cursor.close()
115          return countryArrayList
116      }

120      fun getCountriesCode(code: String): ArrayList<Country> {
121          val countriesList: ArrayList<Country> = ArrayList()
122          val query = "SELECT * FROM $TABLE_COUNTRY WHERE $COLUMN_CODE = $code"
123          val cursor = mDataBase!!.rawQuery(query, null)
124          if (cursor != null) {
125              cursor.moveToFirst()
126              while (!cursor.isAfterLast) {
127                  val id = cursor.getInt(0)
128                  val flag = cursor.getString(1)
129                  val name = cursor.getString(2)
130                  val code = cursor.getString(3)
131                  countriesList.add(Country(id, flag, name, code))
132                  cursor.moveToNext()
133              }
134          }
135          cursor!!.close()
136          return countriesList
137      }
