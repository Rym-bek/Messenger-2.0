38       private lateinit var findUserAdapter: FindUserAdapter

44           inflater: LayoutInflater,
45           container: ViewGroup?,

47       ): View {

62           super.onViewCreated(view, savedInstanceState)

64       }

66   
67       private fun setAdapter()
68       {
69           findUserAdapter = FindUserAdapter(requireContext(), users)
70           viewManager = LinearLayoutManager(requireContext())
71           binding.recyclerView.apply {
72               adapter = findUserAdapter
73               layoutManager = viewManager
74               addItemDecoration(DividerItemDecoration(this.context, DividerItemDecoration.VERTICAL))
75           }
76       }
77   
78       private fun initToolbar()
79       {
80           val navController = findNavController()
81           val appBarConfiguration = AppBarConfiguration(navController.graph)
82           binding.toolbar.setupWithNavController(navController, appBarConfiguration)
83   
84           (requireActivity() as AppCompatActivity).setSupportActionBar(binding.toolbar)
85   
86           binding.toolbar.setNavigationOnClickListener {
87               navController.navigateUp()
88           }
89       }
90   
91       private fun observers()
92       {
93           viewModelGetUsers.users.observe(viewLifecycleOwner)
94           {
95               when (it) {
96                   is BaseResponse.Loading -> {
97   
98                   }
99                   is BaseResponse.Success -> {
100                      users= it.data!!
101                      findUserAdapter.filterList(users)
102                  }
103                  is BaseResponse.Error -> {
104                  }
105              }
106          }
107      }
108  
109      override fun onCreateMenu(menu: Menu, menuInflater: MenuInflater) {
110          menuInflater.inflate(R.menu.menu_search, menu)
111          val searchItem = menu.findItem(R.id.toolbar_searchView)
112          val searchView = searchItem.actionView as SearchView
113          searchView.queryHint = getString(R.string.search)
114          searchView.setOnQueryTextListener(object : SearchView.OnQueryTextListener {
115              override fun onQueryTextSubmit(query: String?): Boolean {
116                  if (query != null && query!="") {
117                      viewModelGetUsers.loadUsers(query.lowercase(), myUid!!)
118                      searchView.clearFocus()
119                  }
120                  return true
121              }
122  
123              override fun onQueryTextChange(newText: String?): Boolean {
124  
125                  return true
126              }
127          })
128      }
129  
130      override fun onMenuItemSelected(menuItem: MenuItem): Boolean {
131          return false
132      }
