22   class FindUserAdapter(private val context:Context, private var users: java.util.ArrayList<UserDialog>)
23       : RecyclerView.Adapter<FindUserAdapter.FindUserAdapterViewHolder>() {
24   
25       class FindUserAdapterViewHolder(var binding:ItemUserBinding) : RecyclerView.ViewHolder(binding.root)
26   
27       override fun onCreateViewHolder(parent: ViewGroup, viewType: Int): FindUserAdapterViewHolder {
28           val binding = ItemUserBinding.inflate(LayoutInflater.from(parent.context), parent, false)
29           return FindUserAdapterViewHolder(binding)
30       }
31   
32       fun filterList(filterllist: ArrayList<UserDialog>) {
33           users = filterllist
34           notifyDataSetChanged()
35       }
36   
37       override fun onBindViewHolder(holder: FindUserAdapterViewHolder, position: Int) {
38           val userDialog: UserDialog = users[position]
39           if(userDialog.firstname!=null)
40           {
41               holder.binding.name.text=userDialog.firstname+" "+userDialog.lastname
42           }
43           else
44           {
45               holder.binding.name.text=null
46           }
47           holder.binding.nickName.text=userDialog.nickname
48   
49           Glide.with(context)
50               .load(userDialog.photo_url)
51               .signature(ObjectKey(userDialog.photo_url.hashCode()))
52               .into(holder.binding.image)
53   
54           holder.itemView.setOnClickListener{ view ->
55               val bundle = bundleOf(
56                   "roomUid" to userDialog.roomUid,
57                   "anotherUid" to userDialog.anotherUid,
58                   "photo_url" to userDialog.photo_url,
59                   "firstname" to userDialog.firstname,
60               )
61               view.findNavController().navigate(R.id.action_fragmentFindUser_to_fragmentUserMessagePage,bundle)
62           }
63       }
64   
65       override fun getItemCount(): Int {
66           return users.size
67       }
