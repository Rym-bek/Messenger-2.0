1    package com.example.messenger.repositories
2    
3    import com.example.messenger.utils.constants.Constants
4    import okhttp3.OkHttpClient
5    import okhttp3.Request
6    import okhttp3.WebSocket
7    import okhttp3.WebSocketListener
8    import javax.inject.Inject
9    import javax.inject.Singleton
10   
11   @Singleton
12   class WebSocketRepository @Inject constructor() {
13       private var webSocket: WebSocket? = null
14   
15       fun connect(listener: WebSocketListener,uid: String) {
16           val client = OkHttpClient()
17           val request = Request.Builder()
18               .url(Constants.WEB_SOCKET_URL)
19               .addHeader("uid",uid)
20               .build()
21           webSocket = client.newWebSocket(request, listener)
22       }
23   
24       fun disconnect() {
25           webSocket?.close(1000, null)
26           webSocket = null
27       }
28   
29       fun send(message: String) {
30           webSocket?.send(message)
31       }
32   }
