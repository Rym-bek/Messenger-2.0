1    package com.example.messenger.viewmodels
2    
3    import androidx.lifecycle.MutableLiveData
4    import androidx.lifecycle.ViewModel
5    import androidx.lifecycle.viewModelScope
6    import com.example.messenger.models.AuthResponse
7    import com.example.messenger.models.ProfileAuth
8    import com.example.messenger.repositories.AuthRepository
9    import com.example.messenger.utils.BaseResponse
10   import dagger.hilt.android.lifecycle.HiltViewModel
11   import kotlinx.coroutines.launch
12   import javax.inject.Inject
13   
14   @HiltViewModel
15   class ViewModelRegistration @Inject constructor(
16       private val repository: AuthRepository
17       ): ViewModel() {
18   
19       val createResponse = MutableLiveData<BaseResponse<AuthResponse>>()
20   
21       fun createAccount(profileAuth: ProfileAuth) {
22           viewModelScope.launch {
23               try {
24                   val response = repository.createProfile(profileAuth)
25                   if (response.code() == 200) {
26                       createResponse.value = BaseResponse.Success(response.body())
27                   } else {
28                       createResponse.value = BaseResponse.Error(response.message())
29                   }
30               } catch (ex: Exception) {
31                   createResponse.value = BaseResponse.Error(ex.message)
32               }
33           }
34       }
35   }
