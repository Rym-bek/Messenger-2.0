1    package com.example.messenger.viewmodels
2    
3    import androidx.lifecycle.MutableLiveData
4    import androidx.lifecycle.ViewModel
5    import androidx.lifecycle.viewModelScope
6    import com.example.messenger.models.UserDialog
7    import com.example.messenger.repositories.MainRepository
8    import com.example.messenger.utils.BaseResponse
9    import dagger.hilt.android.lifecycle.HiltViewModel
10   import kotlinx.coroutines.Dispatchers
11   import kotlinx.coroutines.launch
12   import javax.inject.Inject
13   import kotlin.collections.ArrayList
14   
15   @HiltViewModel
16   class ViewModelAllDialogs @Inject constructor(
17       private val repository: MainRepository
18   ): ViewModel() {
19       //fun getLocationList(): MutableLiveData<ArrayList<Location>> = locationList
20   
21       val dialogs = MutableLiveData<BaseResponse<ArrayList<UserDialog>>>()
22   
23       fun loadDialogs(userUid: String) {
24           viewModelScope.launch(Dispatchers.IO) {
25               try {
26                   val response = repository.getAllDialogs(userUid)
27   
28                   if (response.code() == 200) {
29                       dialogs.postValue(BaseResponse.Success(response.body()))
30                   } else {
31                       dialogs.postValue(BaseResponse.Error(response.message()))
32                   }
33               } catch (ex: Exception) {
34                   BaseResponse.Error(ex.message).toString()
35               }
36           }
37       }
38   
39       fun update(dialogs: ArrayList<UserDialog>)
40       {
41           this.dialogs.postValue(BaseResponse.Success(dialogs))
42       }
43   }
