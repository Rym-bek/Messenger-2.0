31   
32           val body = MultipartBody.Part
33               .createFormData(
34                   name = "image",
35                   filename = file.name,
36                   body = file.asRequestBody())
37   
38           viewModelScope.launch {
39               try {
40                   val response = repository.updateProfilePhoto(body)
41                   if (response.code() == 200) {
42                       uploadPhotoResponse.value = BaseResponse.Success(response.body())
43                   } else {
44                       uploadPhotoResponse.value = BaseResponse.Error(response.message())
45                   }
46               } catch (ex: Exception) {
47                   uploadPhotoResponse.value = BaseResponse.Error(ex.message)
48               }
49           }
50       }
