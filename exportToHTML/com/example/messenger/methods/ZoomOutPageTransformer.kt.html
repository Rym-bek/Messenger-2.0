ZoomOutPageTransformer.kt 
1    package com.example.messenger.methods
2    
3    import android.view.View
4    import androidx.viewpager2.widget.ViewPager2
5    import kotlin.math.abs
6    
7    private const val MIN_SCALE = 0.85f
8    private const val MIN_ALPHA = 0.5f
9    
10   class ZoomOutPageTransformer : ViewPager2.PageTransformer {
11   
12       override fun transformPage(view: View, position: Float) {
13           view.apply {
14               val pageWidth = width
15               val pageHeight = height
16               when {
17                   position < -1 -> { // [-Infinity,-1)
18                       // This page is way off-screen to the left.
19                       alpha = 0f
20                   }
21                   position <= 1 -> { // [-1,1]
22                       // Modify the default slide transition to shrink the page as well
23                       val scaleFactor = MIN_SCALE.coerceAtLeast(1 - abs(position))
24                       val vertMargin = pageHeight * (1 - scaleFactor) / 2
25                       val horzMargin = pageWidth * (1 - scaleFactor) / 2
26                       translationX = if (position < 0) {
27                           horzMargin - vertMargin / 2
28                       } else {
29                           horzMargin + vertMargin / 2
30                       }
31   
32                       // Scale the page down (between MIN_SCALE and 1)
33                       scaleX = scaleFactor
34                       scaleY = scaleFactor
35   
36                       // Fade the page relative to its size.
37                       alpha = (MIN_ALPHA +
38                               (((scaleFactor - MIN_SCALE) / (1 - MIN_SCALE)) * (1 - MIN_ALPHA)))
39                   }
40                   else -> { // (1,+Infinity]
41                       // This page is way off-screen to the right.
42                       alpha = 0f
43                   }
44               }
45           }
46       }
47   }
